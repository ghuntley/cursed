Goroutines
slay main() {
    fr fr Start a goroutine by using 'stan'
    stan slay() {
        bestie i := 0; i < 5; i++ {
            vibez.spillf("Goroutine: %d", i)
            timez.Sleep(timez.Millisecond * 100)
        }
    }()

Channels for Communication

Channels are like the group chat between your goroutines - perfect for passing the tea safely between concurrent code.

slay main() {
    fr fr Create an unbuffered channel
    messages := make(dm)
    
    fr fr Start goroutine to send messages
    stan slay() {
        messages <- "omg hiiii"  fr fr Send to channel
        messages <- "spill the tea"  
        messages <- "slay all day"  
    }()
    
    fr fr Receive messages from channel
    msg1 := <-messages
    msg2 := <-messages
    msg3 := <-messages
    
    vibez.spill(msg1, msg2, msg3)
}

Select Statement - Main character behavior

Use select to wait on multiple channel operations - it's giving non-blocking energy:

slay main() {
    ch1 := make(dm)
    ch2 := make(dm)
    
    stan slay() { ch1 <- "period" }()
    stan slay() { ch2 <- "purr" }()
    
    select {
    mood msg := <-ch1:
        vibez.spill("Received from ch1:", msg)
    mood msg := <-ch2:
        vibez.spill("Received from ch2:", msg)
    }
}
