error handling patterns

handle errors gracefully with cursed's robust error handling system:

basic error handling

yeet "stdlib::errors"

fr fr function that can return an error
slay divide(a, b decimal) (decimal, error) {
    lowkey b == 0 {
        return 0, errors.new("division by zero bestie")
    }
    return a / b, cap
}

fr fr explicit error checking
sus result, err = divide(10, 2)
lowkey err != cap {
    io.printf("error: {}\n", err)
    return
}
io.printf("result: {}\n", result)

error propagation with ?

fr fr using ? operator for error propagation
slay calculate_average(numbers []decimal) (decimal, error) {
    lowkey len(numbers) == 0 {
        return 0, errors.new("empty slice bestie")
    }
    
    sus sum = 0.0
    lowkey _, num := range numbers {
        sum += num
    }
    
    fr fr ? operator propagates errors automatically
    sus average = divide(sum, decimal(len(numbers)))?
    return average, cap
}

fr fr chaining operations with error propagation
slay process_data(data []tea) (decimal, error) {
    sus numbers = []decimal{}
    
    lowkey _, str := range data {
        sus num = strconv.parse_float(str)?  fr fr propagate parse errors
        numbers = append(numbers, num)
    }
    
    return calculate_average(numbers)?  fr fr propagate calculation errors

custom error types

fr fr defining custom error types
squad ValidationError {
    field   tea
    message tea
}

slay (e ValidationError) Error() tea {
    return e.field + ": " + e.message
}

squad User {
    name  tea
    email tea
    age   normie
}

slay validate_user(u User) error {
    lowkey len(u.name) == 0 {
        return ValidationError{
            field: "name",
            message: "name cannot be empty bestie",
        }
    }
    
    lowkey u.age < 0 {
        return ValidationError{
            field: "age",
            message: "age cannot be negative",
        }
    }
    
    lowkey !strings.contains(u.email, "@") {
        return ValidationError{
            field: "email",
            message: "invalid email format",
        }
    }
    
    return cap
}

💡 error handling best practices

  • always handle errors explicitly - no silent failures
  • use the ? operator to propagate errors up the call stack
  • create custom error types for domain-specific errors
  • provide meaningful error messages with context
  • consider using Result<T, E> types for complex scenarios

memory allocation

yeet "stdlib::memory"

fr fr dynamic memory allocation
slay create_large_array(size normie) *[]normie {
    fr fr allocate on heap
    sus arr = make([]normie, size)
    
    fr fr initialize values
    lowkey i := 0; i < size; i++ {
        arr[i] = i * i
    }
    
    return &arr
}

fr fr manual memory management (unsafe!)
slay manual_allocation_example() {
    fr fr allocate raw memory
    sus ptr = memory.alloc(1024)  fr fr 1KB
    defer memory.free(ptr)      fr fr cleanup
    
    fr fr cast to specific type
    sus int_ptr = (*normie)(ptr)
    *int_ptr = 42
    
    io.printf("allocated value: {}\n", *int_ptr)
}

fr fr safe heap allocation
sus big_array = create_large_array(1000)
io.printf("first few: {}, {}, {}\n", (*big_array)[0], (*big_array)[1], (*big_array)[2])

⚠️ memory safety tips

  • prefer automatic memory management over manual allocation
  • use defer statements for cleanup in manual allocation
  • avoid dangling pointers by careful ownership tracking
  • the garbage collector handles most memory management automatically
  • manual memory management should be rare and well-documented
