Overview

CURSED provides powerful concurrency features inspired by Go's goroutines and channels, but with Gen Z flair. The language supports lightweight concurrent execution through goroutines (using the stan keyword) and safe communication via channels (using the dm keyword).

🚀 Goroutines

Lightweight concurrent functions that run independently

💬 Channels

Type-safe communication pipes between goroutines

🔄 Select Statements

Non-blocking channel operations with multiple choices

⚡ High Performance

Efficient runtime with work-stealing scheduler

Goroutines (stan)

Goroutines are lightweight concurrent functions launched with the stan keyword. They run independently and can communicate through channels.

Creating Goroutines

// Basic goroutine creation
stan say_hello("World")

// Goroutine with a closure
stan {
    periodt {
        println("Background task running...")
        sleep(1000) // Sleep for 1 second
    }
}

// Goroutine with parameters
slay process_data(data: String) {
    println("Processing: " + data)
    // Do some work...
}

stan process_data("Important data")

Goroutine Lifecycle

import "sync"

sus counter = 0
sus wg = sync::WaitGroup::new()

// Spawn multiple goroutines
lowkey (sus i = 0; i < 5; i++) {
    wg.add(1)
    stan worker(i, &wg)
    yolo // Yield point for cooperative scheduling
}

slay worker(id: i32, wg: &sync::WaitGroup) {
    println("Worker " + id.to_string() + " starting")
    
    // Simulate work
    lowkey (sus j = 0; j < 3; j++) {
        counter += 1
        sleep(100)
        yolo // Cooperative yield
    }
    
    println("Worker " + id.to_string() + " done")
    wg.done()
}

// Wait for all goroutines to complete
wg.wait()
println("All workers completed. Counter: " + counter.to_string())

Goroutine Patterns

// Fire-and-forget pattern
stan log_event("User logged in")

// Worker pool pattern
sus work_queue = dm make(String, 100) // Buffered channel

// Start workers
lowkey (sus i = 0; i < 4; i++) {
    stan worker(i, work_queue)
}

// Send work
work_queue <- "Task 1"
work_queue <- "Task 2"
work_queue <- "Task 3"

slay worker(id: i32, queue: dm String) {
    periodt {
        bestie task = <-queue {
            println("Worker " + id.to_string() + " processing: " + task)
            // Process task
        } flex {
            println("Worker " + id.to_string() + " no work available")
            break
        }
        yolo
    }
}

Channels (dm)

Channels provide safe communication between goroutines. They're created with the dm keyword and can be buffered or unbuffered.

Creating Channels

// Unbuffered channel (synchronous)
sus messages = dm make(String)

// Buffered channel (asynchronous up to buffer size)
sus numbers = dm make(i32, 10)

// Receive-only channel
sus read_only: dm <-String = messages

// Send-only channel  
sus write_only: dm String<- = messages

Sending and Receiving

sus ch = dm make(String, 5)

// Sending to channel
ch <- "Hello"
ch <- "World"

// Receiving from channel
sus msg1 = <-ch
sus msg2 = <-ch

println(msg1) // "Hello"
println(msg2) // "World"

// Non-blocking receive with bestie/flex
bestie result = <-ch {
    println("Received: " + result)
} flex {
    println("No message available")
}

Channel Communication Patterns

// Producer-Consumer pattern
sus data_channel = dm make(i32, 100)

// Producer goroutine
stan producer(ch: dm i32<-) {
    lowkey (sus i = 0; i < 10; i++) {
        ch <- i
        sleep(100)
        yolo
    }
    close(ch) // Signal completion
}

// Consumer goroutine
stan consumer(ch: dm <-i32) {
    periodt {
        bestie value = <-ch {
            println("Consumed: " + value.to_string())
        } flex {
            println("Channel closed")
            break
        }
        yolo
    }
}

stan producer(data_channel)
stan consumer(data_channel)

Channel Operations

Select Statements

Select statements allow non-blocking operations on multiple channels simultaneously.

sus ch1 = dm make(String)
sus ch2 = dm make(i32)
sus timeout = dm make(bool)

// Timeout goroutine
stan {
    sleep(5000) // 5 seconds
    timeout <- true
}

periodt {
    vibe_check {
        mood msg = <-ch1:
            println("String received: " + msg)
        
        mood num = <-ch2:
            println("Number received: " + num.to_string())
        
        mood <-timeout:
            println("Operation timed out")
            break
            
        basic:
            println("No channels ready")
            sleep(100)
            yolo
    }
}

Channel Closing and Range

sus numbers = dm make(i32, 10)

// Send numbers and close
stan {
    lowkey (sus i = 0; i < 5; i++) {
        numbers <- i
    }
    close(numbers)
}

// Range over channel until closed
bestie num := range numbers {
    println("Number: " + num.to_string())
}

// Check if channel is closed
sus (value, ok) = <-numbers
lowkey (!ok) {
    println("Channel is closed")
}

Directional Channels

// Send-only function parameter
slay sender(ch: dm String<-) {
    ch <- "Message from sender"
}

// Receive-only function parameter
slay receiver(ch: dm <-String) {
    sus msg = <-ch
    println("Received: " + msg)
}

sus bidirectional = dm make(String)
sender(bidirectional)   // Automatically converts to send-only
receiver(bidirectional) // Automatically converts to receive-only

Synchronization Primitives

WaitGroup

import "sync"

sus wg = sync::WaitGroup::new()
sus results = dm make(String, 3)

lowkey (sus i = 0; i < 3; i++) {
    wg.add(1)
    stan fetch_data(i, results, &wg)
}

slay fetch_data(id: i32, results: dm String<-, wg: &sync::WaitGroup) {
    // Simulate API call
    sleep(1000 + (id * 500))
    results <- "Data from source " + id.to_string()
    wg.done()
}

wg.wait()
close(results)

// Collect all results
bestie result := range results {
    println(result)
}

Mutex for Shared State

import "sync"

squad Counter {
    sus value: i32,
    sus mutex: sync::Mutex
}

impl Counter {
    slay new() -> Counter {
        Counter {
            value: 0,
            mutex: sync::Mutex::new()
        }
    }
    
    slay increment(&self) {
        sus lock = self.mutex.lock()
        self.value += 1
        // Lock automatically released when 'lock' goes out of scope
    }
    
    slay get(&self) -> i32 {
        sus lock = self.mutex.lock()
        self.value
    }
}

sus counter = Counter::new()

// Multiple goroutines incrementing safely
lowkey (sus i = 0; i < 5; i++) {
    stan {
        lowkey (sus j = 0; j < 100; j++) {
            counter.increment()
            yolo
        }
    }
}

sleep(1000) // Wait for goroutines
println("Final count: " + counter.get().to_string())

Common Concurrency Patterns

Worker Pool

squad WorkerPool {
    sus jobs: dm T<-,
    sus results: dm T->,
    sus workers: i32
}

impl WorkerPool {
    slay new(worker_count: i32, buffer_size: i32) -> WorkerPool {
        WorkerPool {
            jobs: dm make(T, buffer_size),
            results: dm make(T, buffer_size),
            workers: worker_count
        }
    }
    
    slay start(&self, process_fn: F) 
    where F: Fn(T) -> T + Send + Sync + 'static {
        lowkey (sus i = 0; i < self.workers; i++) {
            sus jobs = self.jobs.clone()
            sus results = self.results.clone()
            
            stan {
                periodt {
                    bestie job = <-jobs {
                        sus result = process_fn(job)
                        results <- result
                    } flex {
                        break
                    }
                    yolo
                }
            }
        }
    }
    
    slay submit(&self, job: T) {
        self.jobs <- job
    }
    
    slay get_result(&self) -> T {
        <-self.results
    }
}

// Usage
sus pool = WorkerPool::new(4, 100)
pool.start(|x: i32| x * x) // Square function

// Submit jobs
lowkey (sus i = 1; i <= 10; i++) {
    pool.submit(i)
}

// Collect results
lowkey (sus i = 0; i < 10; i++) {
    sus result = pool.get_result()
    println("Result: " + result.to_string())
}

Pipeline Pattern

// Pipeline: numbers -> squares -> sums
slay pipeline_demo() {
    sus numbers = dm make(i32, 10)
    sus squares = dm make(i32, 10)
    sus sums = dm make(i32, 10)
    
    // Stage 1: Generate numbers
    stan {
        lowkey (sus i = 1; i <= 5; i++) {
            numbers <- i
        }
        close(numbers)
    }
    
    // Stage 2: Square numbers
    stan {
        bestie num := range numbers {
            squares <- num * num
        }
        close(squares)
    }
    
    // Stage 3: Running sum
    sus running_sum = 0
    stan {
        bestie square := range squares {
            running_sum += square
            sums <- running_sum
        }
        close(sums)
    }
    
    // Collect final results
    bestie sum := range sums {
        println("Running sum: " + sum.to_string())
    }
}

Fan-In/Fan-Out Pattern

// Fan-out: distribute work to multiple goroutines
slay fan_out(input: dm <-T, worker_count: i32) -> Vec {
    sus outputs = Vec::new()
    
    lowkey (sus i = 0; i < worker_count; i++) {
        sus output = dm make(T, 10)
        outputs.push(output.clone())
        
        stan {
            periodt {
                bestie item = <-input {
                    output <- item
                } flex {
                    close(output)
                    break
                }
                yolo
            }
        }
    }
    
    outputs
}

// Fan-in: merge multiple channels into one
slay fan_in(inputs: Vec) -> dm <-T {
    sus output = dm make(T, 10)
    sus wg = sync::WaitGroup::new()
    
    bestie input := inputs {
        wg.add(1)
        stan {
            bestie item := range input {
                output <- item
            }
            wg.done()
        }
    }
    
    // Close output when all inputs are done
    stan {
        wg.wait()
        close(output)
    }
    
    output
}

// Usage
sus source = dm make(String, 100)
sus workers = fan_out(source, 3)
sus merged = fan_in(workers)

// Send work
stan {
    lowkey (sus i = 0; i < 10; i++) {
        source <- "Task " + i.to_string()
    }
    close(source)
}

// Collect results
bestie result := range merged {
    println("Processed: " + result)
}

Error Handling in Concurrent Code

Error Channels

squad WorkResult {
    sus data: Option,
    sus error: Option
}

slay concurrent_work() -> Result, String> {
    sus results = dm make(WorkResult, 10)
    sus wg = sync::WaitGroup::new()
    
    // Spawn workers
    lowkey (sus i = 0; i < 3; i++) {
        wg.add(1)
        stan risky_worker(i, results.clone(), &wg)
    }
    
    // Close results channel when done
    stan {
        wg.wait()
        close(results)
    }
    
    // Collect results and handle errors
    sus success_results = Vec::new()
    bestie result := range results {
        bestie err = result.error {
            return Err(err)
        }
        
        bestie data = result.data {
            success_results.push(data)
        }
    }
    
    Ok(success_results)
}

slay risky_worker(id: i32, results: dm WorkResult<-, wg: &sync::WaitGroup) {
    // Simulate work that might fail
    lowkey (id == 1) {
        results <- WorkResult {
            data: None,
            error: Some("Worker 1 failed".to_string())
        }
    } flex {
        results <- WorkResult {
            data: Some("Success from worker " + id.to_string()),
            error: None
        }
    }
    wg.done()
}

Context and Cancellation

import "context"

slay cancellable_operation(ctx: &context::Context) -> Result {
    sus result_ch = dm make(String, 1)
    sus error_ch = dm make(String, 1)
    
    // Start long-running operation
    stan {
        // Simulate slow work
        sleep(5000)
        result_ch <- "Operation completed"
    }
    
    // Wait for completion or cancellation
    vibe_check {
        mood result = <-result_ch:
            Ok(result)
        
        mood error = <-error_ch:
            Err(error)
        
        mood <-ctx.done():
            Err("Operation cancelled".to_string())
    }
}

// Usage with timeout
sus ctx = context::with_timeout(context::background(), 3000) // 3 seconds

bestie result = cancellable_operation(&ctx) {
    println("Success: " + result)
} yikes err {
    println("Error: " + err)
}

Performance & Best Practices

Performance Considerations

🚀 Goroutine Performance

  • Lightweight: Each goroutine uses ~64KB of stack memory
  • Fast Creation: Goroutine creation takes ~1μs
  • Cooperative Scheduling: Use yolo for yield points
  • Work Stealing: Automatic load balancing across CPU cores

Channel Performance

// Buffer size affects performance
sus unbuffered = dm make(i32)     // Synchronous, blocks on send/receive
sus small_buffer = dm make(i32, 10)   // Good for small bursts
sus large_buffer = dm make(i32, 1000) // Good for high throughput

// Measure channel throughput
import "time"

slay benchmark_channel_throughput() {
    sus ch = dm make(i32, 1000)
    sus count = 100000
    
    sus start = time::now()
    
    // Producer
    stan {
        lowkey (sus i = 0; i < count; i++) {
            ch <- i
            yolo // Yield every iteration for fairness
        }
        close(ch)
    }
    
    // Consumer
    sus received = 0
    bestie _ := range ch {
        received += 1
    }
    
    sus duration = time::since(start)
    sus throughput = (count as f64) / duration.as_secs_f64()
    
    println("Throughput: " + throughput.to_string() + " messages/sec")
}

Best Practices

✅ Do

  • Use channels for communication between goroutines
  • Close channels when done sending
  • Use buffered channels for performance-critical paths
  • Add yolo yield points in long-running loops
  • Use WaitGroup for synchronizing goroutine completion
  • Prefer select statements for non-blocking operations

❌ Don't

  • Share memory without synchronization (use Mutex or channels)
  • Create too many goroutines (use worker pools instead)
  • Block forever on channel operations without timeouts
  • Forget to close channels in producer goroutines
  • Use global variables in concurrent code without protection

// Detect potential deadlocks
slay deadlock_detection() {
    import "context"
    
    sus ctx = context::with_timeout(context::background(), 10000) // 10 seconds
    sus ch = dm make(i32)
    
    vibe_check {
        mood value = <-ch:
            println("Received: " + value.to_string())
        
        mood <-ctx.done():
            println("Potential deadlock detected - no message received")
    }
}

Memory Management

🧠 GC Integration

CURSED's garbage collector is goroutine-aware:

  • Stack Scanning: GC scans goroutine stacks for live objects
  • Safe Points: Collection only occurs at safe points
  • Concurrent Collection: GC runs concurrently with goroutines
  • Low Pause Times: Typically <10ms for moderate goroutine counts
